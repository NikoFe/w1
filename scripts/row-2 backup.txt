function buttonFocus(number) {
  for (i = 0; i < buttons.length; i++) {
    if (i == number) {
      buttons[i].style.backgroundColor = "#cfe2fa";
    } else {
      buttons[i].style.backgroundColor = "white";
    }
  }
}
